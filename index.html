        let currentQuestion = 0;

        function startQuiz() {

            document.getElementById('welcome-screen').classList.add('hidden');

            document.getElementById('quiz-screen').classList.remove('hidden');

            showQuestion();

        }

 

        function showQuestion() {

            const q = questions[currentQuestion];

            document.getElementById('progress-text').innerText = `問題 ${currentQuestion + 1} / 10`;

            document.getElementById('progress-bar').style.width = `${(currentQuestion + 1) * 10}%`;

            document.getElementById('question-text').innerText = q.q;

           

            const container = document.getElementById('options-container');

            container.innerHTML = '';

           

            q.options.forEach((opt, index) => {

                const btn = document.createElement('button');

                btn.className = "option-card w-full text-left p-5 rounded-2xl bg-gray-50 border-2 border-transparent transition-all duration-200 hover:border-red-500 font-medium text-gray-700";

                btn.innerText = opt;

                btn.onclick = () => selectOption(index);

                container.appendChild(btn);

            });

        }

 

        function selectOption(index) {

            const weight = questions[currentQuestion].weights[index];

            const choiceText = questions[currentQuestion].options[index];

            userScores[weight]++;

            historyChoices.push({ question: questions[currentQuestion].q, answer: choiceText });

 

            if (currentQuestion < questions.length - 1) {

                currentQuestion++;

                showQuestion();

            } else {

                calculateResult();

            }

        }

            document.getElementById('quiz-screen').classList.add('hidden');

           

            let winner = 'R';

            for (const key in userScores) {

                if (userScores[key] > userScores[winner]) winner = key;

            }

 

            const pool = snacks[winner];

            const score = userScores[winner];

           

            if (score >= 4) {

                finalSnack = pool[0];

            } else if (score === 3) {

                finalSnack = pool[1] || pool[0];

            } else {

                finalSnack = pool[pool.length - 1];

            }

 

            displayResult(finalSnack);

        }

 

        async function fetchGeminiAnalysis() {

            const analysisBtn = document.getElementById('ai-analysis-btn');

            const aiContent = document.getElementById('ai-content');

           

            analysisBtn.disabled = true;

            aiContent.classList.remove('hidden');

            aiContent.innerHTML = `<div class="p-4 rounded-xl loading-shimmer h-24 w-full"></div>`;

           

            const systemPrompt = `你是一個資深的心理學家和老牌港式零食專家。用戶剛玩完一個心理測驗，

            結果是「${finalSnack.name}」。請根據用戶之前的 10 個具體選擇，

            提供一段充滿地道廣東話、風趣幽默且深層的性格分析報告。

            1. 基於具體答題內容的深層解讀。

            2. 該人格在辦公室/學校的「隱藏人設」。

            3. 給用戶的一句「零食格言」。

            請直接輸出內容，不要包含 Markdown 符號。`;

 

            const userQuery = `測驗結果：${finalSnack.name}。答題歷程：${JSON.stringify(historyChoices)}`;

 

            try {

                let retries = 0;

                let resultText = "";

               

                while (retries < 5) {

                    try {

                        const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {

                            method: 'POST',

                            headers: { 'Content-Type': 'application/json' },

                            body: JSON.stringify({

                                contents: [{ parts: [{ text: userQuery }] }],

                                systemInstruction: { parts: [{ text: systemPrompt }] }

                            })

                        });

                        const data = await response.json();

                        resultText = data.candidates?.[0]?.content?.parts?.[0]?.text;

                        if (resultText) break;

                    } catch (e) {

                        retries++;

                        await new Promise(r => setTimeout(r, Math.pow(2, retries) * 1000));

                    }

                }

 

                if (!resultText) throw new Error("API Failed");

               

                aiContent.innerHTML = `<div class="p-4 bg-white/60 rounded-xl text-sm leading-relaxed text-gray-700 italic border-l-4 border-red-400">${resultText.replace(/\n/g, '<br>')}</div>`;

                analysisBtn.innerText = "✨ AI 解析完成";

            } catch (err) {

                aiContent.innerHTML = `<div class="p-4 bg-red-50 text-red-500 rounded-xl text-sm italic">哎呀，AI 呢排可能食滯咗，解析唔到住...</div>`;

                analysisBtn.disabled = false;

            }

        }

            document.body.removeChild(dummy);
